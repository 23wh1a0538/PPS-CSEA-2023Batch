#include <stdio.h>
#include<stdbool.h>
bool binary_search(int[],int, int, int);
int main() {
   int n;
   bool result;
   printf("enter value of n");
   scanf("%d",&n);
   int a[n],key,i;
   for(i=0;i<n;i++)
       scanf("%d",&a[i]);
   scanf("%d",&key);
   result = binary_search(a,key,0,n-1);
   if(result==1)
        printf("element is found");
    else
        printf("element is not found");
}
bool binary_search(int a[],int key,int low, int high){
    int mid;
    while(low<=high){
        mid=(low+high)/2;
        if(a[mid]==key)
            return 1;
        if(a[mid]<key)
            low=mid+1;
        else
            high=mid-1;
    }
    return 0;
}
