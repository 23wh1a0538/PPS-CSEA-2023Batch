void selection_sort(int arr[],int n){
    int min_index,i,j;
    for(i=0;i<n-1;i++){
        min_index = i;
        for(j=1;j<n;j++){
            if(arr[min_index]>arr[j])
                min_index = j;
        }
        if(min_index!=i)
        swap( arr[i], arr[min_index]);
    }
}
void swap(int *n1,int *n2){
    int temp;
    temp = *n1;
    *n1 = *n2;
    *n2 = temp;
}
